class Solution {
    public int[][] construct2DArray(int[] year2021, int m, int n) {
        if (year2021.length != m * n) return new int[0][0];
        int[][] year2022 = new int[m][n];
        for (int i = 0, idx = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                year2022[i][j] = year2021[idx++];
            }
        }
        return year2022;
    }
}

##---------------------- 原理跟第二种方法类似，细节做了修改 -----------------------------
    def sumEvenAfterQueries(self, A: List[int], queries: List[List[int]]) -> List[int]:
        nsum = sum(i for i in A if i & 1 == 0)
        res = []
        
        for value, key in queries:
            if A[key] & 1 == 0:         # 原来为偶数
                if value & 1 == 0:      # 增加的值为偶数，和为偶数
                    nsum += value
                else:                   # 增加的值为奇数，和为奇数
                    nsum -= A[key]
            else:                       # 原来为奇数
                if value & 1:           # 增加的值为奇数，和为偶数
                    nsum += A[key] + value
            res.append(nsum)
            A[key] += value
        
        return res

class Solution {
public:
    vector<int> findKDistantIndices(vector<int>& nums, int key, int k) {
        vector<int> res;
        // j è¡¨ç¤ºç°åœ¨æœªç¡®å®šæ˜¯å¦èƒ½æ”¾å…¥ res çš„ç¬¬ä¸€ä¸ªæ•°
        for(int i = 0 , j = 0;i < nums.size() && j < nums.size();i ++) {
            if(nums[i] == key) {
                j = max(i - k,j);
                while(j < min(i + k + 1,(int) nums.size())) res.push_back(j ++);
            }
        }
        return res;
    }
};

class Solution:
    def countPoints(self, points: List[List[int]], queries: List[List[int]]) -> List[int]:
        l = len(queries)
        ans = [0]*l
        for idx in range(l):
            m,n,r = queries[idx]
            for x,y in points:
                if abs(m -x + (n-y)*(1j)) <= r:
                    ans[idx] += 1
        return ans

class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int m = matrix.size(), n = matrix[0].size();
        for(int i = 0; i < m; ++i)
            for(int j = 1; j < n; ++j)
                matrix[i][j] ^= matrix[i][j-1];

        for(int j = 0; j < n; ++j)
            for(int i = 1; i < m; ++i)
                matrix[i][j] ^= matrix[i-1][j];

        
        priority_queue<int, vector<int>, greater<int>> q;
        for(auto& vec : matrix)
        {
            for(auto& x : vec)
            {
                //cout << x << " ";
                if(q.size() < k)
                    q.push(x);
                else if(x > q.top())
                {
                    q.pop();
                    q.push(x);
                }
            }
            //cout << endl;
        }
        return q.top();
    }
};

        n = len(matrix[0])
        for i in range(m):

                matrix[i][j] ^= matrix[i-1][j];
        ans = []
        for i in range(m):
            for j in range(n):
                ans.append(matrix[i][j])
        ans.sort()
        return ans[len(ans) - k]

class Solution {
public:
    int kthLargestValue(vector<vector<int>>& matrix, int k) {
        int m = matrix.size(), n = matrix[0].size();
        vector<vector<int>> dp(m+1, vector<int>(n+1));
        priority_queue<int, vector<int>, greater<int>> q;
        for(int i = 1; i <= m; ++i)
        {
            for(int j = 1; j <= n; ++j)
            {
                dp[i][j] = dp[i-1][j]^dp[i][j-1]^dp[i-1][j-1]^matrix[i-1][j-1];
                if(q.size() < k) q.push(dp[i][j]);
                else if(dp[i][j] > q.top())
                {
                    q.pop();
                    q.push(dp[i][j]);
                }
            }
        }
        return q.top();
    }
};
